fn meta_condition(x: u64) -> bool {
    x <= 5
}

fn condition(x: u64) -> bool
    requires meta_condition(x)
{
    1 <= x
}

fn helper(x: u64) -> u64
    requires meta_condition(x), condition(x)
    sees condition, meta_condition
{
    x + x
}

fn main()
    sees condition, meta_condition
{
    helper(5);
}
